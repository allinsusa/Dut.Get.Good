ALTER PROCEDURE [dbo].[ClassAbi_GetAllClassAbilities]
AS
BEGIN
    SELECT ca.Id AS 'ClassAbilityId',
           ca.ClassId,
           c.[ClassDescription],
           ca.AttributeId,
           a.AttributeDescription,
           ca.AbilityDescription
    FROM   [dbo].[AppClassAbility] AS ca
           INNER JOIN
           [dbo].[AppClass] AS c
           ON c.Id = ca.ClassId
           INNER JOIN
           [dbo].[AppAttribute] AS a
           ON a.Id = ca.AttributeId;
END

GO
